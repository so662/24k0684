    PRINT "Floor 5" 
ELSE IF n = 1011111
    PRINT "Floor 6" 
ELSE IF n = 1110000 
    PRINT "Floor 7" 
ELSE IF n = 1111111
    PRINT "Floor 8"

    PRINT "Floor 9" 
ELSE 
    PRINT "INVALID" 
END
